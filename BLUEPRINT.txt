swarmbotics-sim-devops/
├── simulation-environment/             # SITL / HITL Simulation (Ignition/Gazebo)
│   ├── models/                         # Virtual representations of physical assets
│   │   └── fireant-ugv/
│   │       ├── model.sdf               # Physics, kinematics, and sensor definitions
│   │       └── meshes/                 # 3D visual and collision geometry
│   ├── worlds/                         # Tactical testing environments
│   │   ├── urban-canyon.sdf            # Simulates GPS-denied environments (multipath errors)
│   │   └── contested-plains.sdf        # Open terrain with simulated adversary EW jammers
│   ├── launch-scripts/                 # Orchestration for swarm-scale testing
│   │   ├── spawn-swarm.py              # Script to instantiate 100+ UGVs with unique namespaces
│   │   └── hitl-bridge.launch.py       # Routes simulated sensor data into a physical edge compute board
│   └── network-sim/                    # Simulates the chaotic realities of tactical comms
│       └── mininet-mesh-sim.py         # Artificially injects latency, packet loss, and jamming
│
├── ci-cd-pipelines/                    # Continuous Integration & Automated Testing
│   ├── .github/workflows/              # (or GitLab CI equivalents)
│   │   ├── build-ros-edge.yml          # Compiles the Edge Layer C++ code and checks formatting
│   │   ├── headless-sim-tests.yml      # Runs automated behavior tests in Gazebo without a GUI
│   │   └── build-c2-containers.yml     # Builds and pushes Docker images for the C2 Backend
│   └── behavior-test-suites/           # Automated validation of the "Hive Mind"
│       ├── test-flanking-logic.py      # Asserts the swarm successfully surrounds a target in sim
│       └── test-comms-failover.py      # Asserts the UGVs switch to SATCOM when the mesh drops
│
└── ota-update-system/                  # Over-The-Air (OTA) Updates for the global fleet
    ├── update-server/                  # Fleet management backend (e.g., Mender.io or custom)
    │   ├── fleet-tracker.go            # Tracks the current software/AI version of every UGV
    │   └── campaign-scheduler.go       # Manages staged rollouts (e.g., update 10% of the swarm first)
    ├── edge-client/                    # The daemon running directly on the Ubuntu Core UGV
    │   ├── ota-agent.rs                # Ultra-reliable Rust client that downloads patches
    │   └── a-b-partition-manager.sh    # Manages dual-bank OS partitions for bulletproof rollbacks
    └── crypto-security/                # Zero-trust update verification
        ├── key-generation.sh           # Generates asymmetric keys for the CI pipeline
        └── verify-payload.py           # Forces the UGV to verify the update's signature before installing
